// Class NcBoost
// Perform various Lorentz transformations.
//
// Example :
// =========
//
// Float_t a[3]={0.1,0.2,0.3};
// Float_t ea[3]={0.01,0.02,0.03};
// Nc3Vector beta;
// beta.SetVector(a,"car");
// beta.SetErrors(ea,"car");
//
// NcBoost b1;
// b1.SetBeta(beta);
// b1.Data();
//
// Float_t b[4]={14,1,2,3};
// Float_t eb[4]={1.4,0.1,0.2,0.3};
// Nc4Vector p;
// p.SetVector(b,"car");
// p.SetErrors(eb,"car");
// Nc4Vector pprim=b1.Boost(p);
// p.Data();
// pprim.Data();
//
// p=b1.Inverse(pprim);
// pprim.Data();
// p.Data();
//
// Float_t c[4]={5,0,0,4};
// Float_t ec[4]={0.5,0,0,0.4};
// Nc4Vector q;
// q.SetVector(c,"car");
// q.SetErrors(ec,"car");
//
// NcBoost b2;
// b2.Set4Momentum(q);
// b2.Data("sph");
//
