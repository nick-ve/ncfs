422{
454
455 if (ndf <= 0) return 0; // Set CL to zero in case ndf<=0
456
457 if (chi2 <= 0.)

459 if (chi2 < 0.)
460 {
461 return 0;
462 }
463 else
464 {
465 return 1;
466 }
467 }

481 if (q>0.) v=0.5*(1.-Erf(q/sqrt(2.)));

490 Double_t x=chi2/2.;
