320 // Create output tree if necessary

341 // Some output for the user's convenience

343 cout << " *" << ClassName() << "::Exec* Overview of scheduled processing." << endl;
344 cout << " ***" << endl;
345 cout << " *** Start processing of job " << GetName() << " ***" << endl;
346 cout << " ***" << endl;

348 cout << " Print frequency : " << fPrintfreq << " (<=0 means no printout)" << endl;
349 cout << " Print level : " << fPrintlevel << endl;

351 {
352 cout << " RnoEvent output file : " << fOutfile->GetName() << endl;
353 cout << " Output characteristics : splitlevel = " << fSplit << " buffersize = " << fBsize << endl;
354 cout << " Required event selection level interval for output : [" << fMinSelectLevel << "," << fMaxSelectLevel << "]" << endl;
355 }

359
360 // The number of entries in the input chain
361 Int_t nen=fData->GetEntries();
362
363 // Check for the maximum number of events to be processed
364 if (fMaxevt>-1 && nen>fMaxevt) nen=fMaxevt;
365
367 // Variables in the RNO-G data tree //
369
370 // Header info
371 Int_t run=0;
372 Int_t event=0;
373 Int_t station=0;
374 Double_t trigtime=0;
375
376 // DAQ info
377 NcDevice daq("DAQ","DAQ status");
378 daq.AddNamedSlot("RADIANT-update");
379 daq.AddNamedSlot("FLOWER-update");
380 daq.AddNamedSlot("Sampling-rate");
381
382 // Trigger info
383 NcTagger trigger("Trigger","Trigger tags");
384
385 // The waveforms of all channels are provided as Short_t radiant_data[24][2048]
386 // but will be readout linearly and stored as an NcSample for each channel, as indicated below.
387 NcSample signal("Signals","Radiant signals");
388 signal.SetStoreMode(1);
389
390 // The pedestals of all channels are provided as UShort_t pedestals[24][4096]
391 // but will be readout linearly and stored as an NcSample for each channel, as indicated below.
392 NcSample pedestal("Pedestals","Pedestal values");
393 pedestal.SetStoreMode(1);
394
396 // Loop over the entries in the input data chain //
398
399 TTree* tx=0; // Pointer to the current Tree in the Chain
400 TTree* txold=0; // Pointer to the previous Tree (if any) in the Chain
401 TObjArray* leaves=0; // Array with pointers to the leaves of the current Tree
402 Int_t nleaves=0;
403 TLeaf* lx=0;
404 TLeaf* lradiant=0; // Pointer to the Radiant waveform data
405 TLeaf* lpedestal=0; // Pointer to the Pedestal waveform data
406 Int_t idx=0;
407 Int_t nsamples=0; // The sampling buffer length
408 NcDevice* devx=0;
409 Int_t idsample=0; // Index for the NcSample in the NcDevice storage
410 TString name="";
411 TString namex1="";
412 TString namex2="";
413 TString namex3="";
414 Float_t value=0;
415 Int_t ivalue=0;
416 Bool_t flag=kFALSE;
417 Int_t iradiant=0; // RADIANT firmware update sequence number
418 Int_t iflower=0; // FLOWER firmware update sequence number
419 Float_t fsample; // The DAQ sampling rate in Hz
420 Int_t nwritten=0; // The number of events written to output
421 Int_t perc=0; // Percentage of the number of processed events
422
423 for (Int_t ient=0; ient<nen; ient++)
424 {
425 // Reset the detector structure
426 det.Reset();
427 idsample=0;
428
429 fData->GetEntry(ient);
430
431 tx=fData->GetTree();
432 if (tx!=txold)
433 {
434 leaves=tx->GetListOfLeaves();
435 nleaves=0;
436 if (leaves) nleaves=leaves->GetEntries();
437 txold=tx;
438 }
439
440 // Loop over all the leaves and extract the relevant data for this entry.
441 // This approach makes the functionality independent of the Tree/Branch structure.
442 run=0;
443 event=0;
444 station=0;
445 trigtime=0;
446 nsamples=0;
447 trigger.Reset();
448 lradiant=0;
449 lpedestal=0;
450 for (Int_t ileaf=0; ileaf<nleaves; ileaf++)
451 {
452 lx=(TLeaf*)leaves->At(ileaf);
453
454 if (!lx) continue;
455
456 name=lx->GetName();
457
458 // Header data
459 if (!run && name=="run_number") run=lx->GetValue();
460 if (!event && name=="event_number") event=lx->GetValue();
461 if (!station && name=="station_number") station=lx->GetValue();
462 if (!trigtime && name=="trigger_time") trigtime=lx->GetValue();
463 if (!trigtime && name=="when") trigtime=lx->GetValue(); // In case of a pedestal data file
464 if (!nsamples && name=="buffer_length") nsamples=lx->GetValue();
465
466 // Trigger data
467 if (name.Contains("trigger_info"))
468 {
469 name.ReplaceAll("trigger_info.","");
470 value=lx->GetValue();
471 ivalue=TMath::Nint(value);
472 namex1="";
473 namex2="";
474 namex3="";
475
476 // The low threshold time window and number of coincidences
477 if (name.Contains("lt_info.window")) namex1="lt-window";
478 if (name.Contains("lt_info.num_coinc")) namex1="lt-ncoinc";
479
480 if (namex1!="")
481 {
482 trigger.AddNamedSlot(namex1);
483 trigger.SetSignal(value,namex1);
484 continue;
485 }
486
487 // The various radiant (=surface) time windows and number of coincidences
488 if (name.Contains("radiant_info.RF_window"))
489 {
490 namex1="LPDA-up-window";
491 namex2="LPDA-down-window";
492 value=lx->GetValue(0);
493 trigger.AddNamedSlot(namex1);
494 trigger.SetSignal(value,namex1);
495 value=lx->GetValue(1);
496 trigger.AddNamedSlot(namex2);
497 trigger.SetSignal(value,namex2);
498 continue;
499 }
500 if (name.Contains("radiant_info.RF_ncoinc"))
501 {
502 namex1="LPDA-up-ncoinc";
503 namex2="LPDA-down-ncoinc";
504 value=lx->GetValue(0);
505 trigger.AddNamedSlot(namex1);
506 trigger.SetSignal(value,namex1);
507 value=lx->GetValue(1);
508 trigger.AddNamedSlot(namex2);
509 trigger.SetSignal(value,namex2);
510 continue;
511 }
512
513 // The actual trigger tags
514 if (name.Contains("_trigger"))
515 {
516 // Settings of the various radiant (=surface) triggers
517 if (name.Contains("which_radiant"))
518 {
519 namex1="LPDA-up_trigger";
520 namex2="LPDA-down_trigger";
521 namex3="radiant-unknown_trigger";
522 if (ivalue<-100) // None of the surface triggers
523 {
524 trigger.SetPass(namex1,kFALSE);
525 trigger.SetPass(namex2,kFALSE);
526 trigger.SetPass(namex3,kFALSE);
527 }
528 if (ivalue==-1) // Unknown radiant trigger
529 {
530 trigger.SetPass(namex1,kFALSE);
531 trigger.SetPass(namex2,kFALSE);
532 trigger.SetPass(namex3,kTRUE);
533 }
534 if (ivalue==0) // Only upward surface trigger
535 {
536 trigger.SetPass(namex1,kTRUE);
537 trigger.SetPass(namex2,kFALSE);
538 trigger.SetPass(namex3,kFALSE);
539 }
540 if (ivalue==1) // Only downward surface trigger
541 {
542 trigger.SetPass(namex1,kFALSE);
543 trigger.SetPass(namex2,kTRUE);
544 trigger.SetPass(namex3,kFALSE);
545 }
546 }
547 else
548 {
549 flag=kFALSE;
550 if (ivalue) flag=kTRUE;
551 trigger.SetPass(name,flag);
552 }
553 }
554 } // End of trigger data
555
556 // Pointers to the waveform data
557 if (name=="radiant_data") lradiant=lx;
558 if (name=="pedestals") lpedestal=lx;
559
560 } // End of loop over the leaves
561
562 // Create this station in the detector structure
563 RnoStation* stax=det.GetStation(station,kTRUE);
564
565 if (!stax) break;
566
567 // DAQ info
568 iradiant=2; // The RADIANT update version
569 iflower=0; // The FLOWER update version
570 fsample=3.2e9; // The sampling rate in Hz
571 if (station==11)
572 {
573 if (run<571) iradiant=1;
574 if (run<474) iradiant=0;
575 if (run>=571) iradiant=2;
576 }
577 if (station==21)
578 {
579 if (run<753) iradiant=1;
580 if (run<646) iradiant=0;
581 }
582 if (station==22)
583 {
584 if (run<656) iradiant=1;
585 if (run<574) iradiant=0;
586 }
587
588 daq.SetSignal(iradiant,"RADIANT-update");
589 daq.SetSignal(iflower,"FLOWER-update");
590 daq.SetSignal(fsample,"Sampling-rate");
591
592 stax->AddDevice(daq);
593
594 stax->AddDevice(trigger);
595
596 // Readout the signal waveforms of all Radiant channels
597 if (!nsamples) nsamples=2048;
598 lx=lradiant;
599 if (lx)
600 {
601 idsample++;
602 idx=0;
603 for (Int_t i=0; i<24; i++)
604 {
605 // Access the corresponding channel of this station
606 name="Ch";
607 name+=i;
608 devx=stax->GetDevice(name,kTRUE);
609
610 if (!devx) continue;
611
612 signal.Reset();
613 signal.SetNames("ADC");
614 for (Int_t j=0; j<nsamples; j++)
615 {
616 // Retrieve the value of radiant[i][j]
617 value=lx->GetValue(idx);
618 signal.Enter(value);
619 idx++;
620 }
621 devx->SetSample(&signal,idsample);
622 }
623 }
624
625 // Readout the pedestals of all Radiant channels
626 nsamples=4096;
627 lx=lpedestal;
628 if (lx)
629 {
630 idsample++;
631 idx=0;
632 for (Int_t i=0; i<24; i++)
633 {
634 // Access the corresponding channel of this station
635 name="Ch";
636 name+=i;
637 devx=stax->GetDevice(name,kTRUE);
638
639 if (!devx) continue;
640
641 pedestal.Reset();
642 pedestal.SetNames("ADC");
643 for (Int_t j=0; j<nsamples; j++)
644 {
645 // Retrieve the value of pedestals[i][j]
646 value=lx->GetValue(idx);
647 pedestal.Enter(value);
648 idx++;
649 }
650 devx->SetSample(&pedestal,idsample);
651 }
652 }
653
654 // Transfer the RNO-G data into the RnoEvent structure
655 evt->Reset();
656 evt->SetUnixTime(trigtime,"A");
657 evt->SetRunNumber(run);
658 evt->SetEventNumber(event);
659 evt->SetDetector(det);
660
661 // Invoke all available sub-tasks (if any)
662 CleanTasks();
663 ExecuteTasks(GetName());
664
665 // Provide a printout every "Printfreq" events
666 if (fPrintfreq>0)
667 {
668 if ((!(ient%fPrintfreq) || (ient+1)==nen))
669 {
670 perc=100*(ient+1)/nen;
671 if (fPrintlevel==0 || fPrintlevel==2)
672 {
673 cout << " *** Processed input entry : " << ient << " run : " << run << " event : " << event << " (" << perc << "%)" << endl;
674 }
675 if (fPrintlevel>0)
676 {
677 evt->HeaderData();
678 cout << endl;
679 }
680 }
681 }
682
683 // Write the event to the output file (if the event select level is o.k.)
684 Int_t select=evt->GetSelectLevel();
685 if (select<fMinSelectLevel) continue;
686 if (fMaxSelectLevel>=fMinSelectLevel && select>fMaxSelectLevel) continue;
687
688 if (otree) otree->Fill();
689 nwritten++;
690 } // End of loop over the entries
691
692 // Flush possible memory resident data to the output file
693 if (fOutfile)
694 {
695 fOutfile->Write();
696 cout << endl;
697 cout << " *" << ClassName() << "::Exec* Number of (selected) events written to output : " << nwritten << endl;
698 }
699
700 // Remove the RnoEvent object from the environment
701 // and delete it as well
702 if (evt)
703 {
704 RemoveObject(evt);
705 delete evt;
706 }
707
708 // Delete the input data chain
709 if (fData)
710 {
711 delete fData;
712 fData=0;
713 }
714}
