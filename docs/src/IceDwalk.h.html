  void SelectQvalue(TObjArray& tes,Float_t qmax); // TC selection via Q-value

  void MergeJets(TObjArray& jets,Float_t jangmax,Float_t jdistmax,Int_t jinvol,Int_t jiterate,Float_t qmax); // Jet Merging
